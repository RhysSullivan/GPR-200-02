vec4 guassianBlur(vec2 fragCoord)
{
 	vec2 fLCoord = vec2(floor(fragCoord.x), fragCoord.y);
    vec2 invRes = 1. / iChannelResolution[0].xy;    
   	vec4 a = 	  texture(iChannel0, convertToUV( vec2(fLCoord.x - 2., fLCoord.y), invRes) ); 
    vec4 b = 3. * texture(iChannel0, convertToUV( vec2(fLCoord.x - 1., fLCoord.y), invRes) );     
    vec4 c = 3. * texture(iChannel0, convertToUV( vec2(fLCoord.x + 1., fLCoord.y), invRes) ); 
    vec4 d = 	  texture(iChannel0, convertToUV( vec2(fLCoord.x + 2., fLCoord.y), invRes) );     
    return 0.125 * (a + b + c + d);
    
}

void mainImage( out vec4 fragColor, in vec2 fragCoord )
{      	 
    fragColor = guassianBlur(fragCoord);
}
